Line Types

init <- par(no.readonly = TRUE)
par(mfrow = c(3, 2))
values <- 1:6
for (i in values) {
    plot(data, type = "l", lty = i, col ="red")
    mtext(paste("lty = ", i), side = 3, col = "blue", line = 1)
}
